55 static rd_status_t store_block (const app_log_record_t * const record)
56 {
57  static uint8_t record_idx = 0;
58  uint8_t num_tries = 0;
59  uint32_t end_timestamp = m_log_input_block.end_timestamp_s;
60  rd_status_t err_code = RD_SUCCESS;
61 
62  do
63  {
64  err_code = rt_flash_free (APP_FLASH_LOG_FILE,
65  (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8U) + record_idx);
66 
67  // It's not a problem if there wasn't old block to erase.
68  if (RD_SUCCESS == err_code)
69  {
70  LOG ("LOG: Erasing old block\r\n");
71  }
72 
73  err_code &= ~RD_ERROR_NOT_FOUND;
74 
75  while (rt_flash_busy())
76  {
77  ri_yield();
78  }
79 
80  err_code |= rt_flash_gc_run ();
81 
82  while (rt_flash_busy())
83  {
84  ri_yield();
85  }
86 
87  err_code |= rt_flash_store (APP_FLASH_LOG_FILE,
88  (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8U) + record_idx,
89  &m_log_input_block, sizeof (m_log_input_block));
90 
91  while (rt_flash_busy())
92  {
93  ri_yield();
94  }
95 
96  RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
97  // Erase another block and try again if there was error.
98  num_tries++;
99  record_idx++;
100  record_idx = record_idx % APP_FLASH_LOG_DATA_RECORDS_NUM;
101  } while ( (RD_SUCCESS != err_code) && (num_tries < APP_FLASH_LOG_DATA_RECORDS_NUM));
102 
103  memset (&m_log_input_block, 0, sizeof (m_log_input_block));
104  m_log_input_block.start_timestamp_s = end_timestamp;
105  return err_code;
106 }
